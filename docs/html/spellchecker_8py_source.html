121  def edit_distance_1(self, word):
122  """
123  Compute all strings that are one edit away from `word` using only the letters in the corpus.
124 
125  Args:
126  word (str): The word for which to calculate the edit distance.
127  Returns:
128  set: The set of strings that are edit distance two from the provided word.
129  """
130  letters = self._word_frequency.letters
131  splits = [(word[:i], word[i:]) for i in range(len(word) + 1)]
132  deletes = [L + R[1:] for L, R in splits if R]
133  transposes = [L + R[1] + R[0] + R[2:] for L, R in splits if len(R) > 1]
134  replaces = [L + c + R[1:] for L, R in splits if R for c in letters]
135  inserts = [L + c + R for L, R in splits for c in letters]
136  return set(deletes + transposes + replaces + inserts)
