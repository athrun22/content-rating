34 def parse_pdf(filename):

36  This function parses a pdf file by parsing the file given by the filename and returning a string of the content in

38  :param filename: The pdf file to parse.
39  :return: A string containing the pdf file's contents.
40  """

43  page_objects = []
44  full_text = ""
45  filtered_text = ""

47  page_objects.append(pdf_reader.getPage(x))
48  for page_object in page_objects:
49  full_text += page_object.extractText()
50  for line in full_text.split('\n'):
